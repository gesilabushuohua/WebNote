6. **JS 基础**，函数传参，按值传递

   ```javascript
   function test1(a){
     // 重新分配地址  
     a =  2;
   }
   
   function test2(obj1){
     // obj1 地址没有改变  
     obj1.a = 2;
   }
   
   function test3(obj2){
     obj2.b = 2;  
     // obj2 内存指向改变 
     obj2 = {b:3};
   }
   
   let a = 0;
   let obj1 = { a:0};
   let obj2 = { b:0};
   
   test1(a);
   test2(obj1);
   test3(obj2);
   
   console.log({a,obj1,obj2}); // {a:0, obj1:{a:2},obj2:{b:2}}
   
   /* 证明 test3 obj 是按值传递 */
   ```
